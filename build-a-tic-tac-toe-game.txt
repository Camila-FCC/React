** start of index.jsx **

const { useState } = React;

export function Board() {

  const [board, setBoard] = useState(Array(9).fill(null));

  const [isX, setIsX] = useState(true);

  
  const winners = board => {
  
    const winLines = [
      [0, 1, 2], 
      [3, 4, 5], 
      [6, 7, 8], 
      [0, 3, 6], 
      [1, 4, 7], 
      [2, 5, 8], 
      [0, 4, 8], 
      [2, 4, 6]
    ];

    for(let i = 0; i < winLines.length; i++) {

      const [a, b, c] = winLines[i];

      if(board[a] && board[a] === board[b] && board[a] === board[c]) {

        return board[a];
      }
    }
    
    return null;
  }


  const handleClick = i => {

    //to disable btn once it was clicked or when we have a winner:
    if(board[i] || winners(board)) {
      return;
    }

    const copy = [...board];

    if(isX) {
      copy[i] = "X";
    }
    else {
      copy[i] = "O";
    }
    
    setBoard(copy);
    setIsX(!isX);
  }

  return (
  <>
    <h1>{"Tic Tac Toe"}</h1>
    <div className="board">
      {board.map((_, i) => <div className="board-square" key={i}><button key={i} className="square" onClick={() => handleClick(i)} style={{
        
        backgroundColor: winners(board) ? "yellow" : "#000221",

        color: winners(board) ? "darkBlue" : "aquamarine"

      }}>{board[i]}</button></div>)}
    </div>
    <button id="reset" onClick={() => {
      setBoard(Array(9).fill(null));
      setIsX(true);
      }}>
      {"Reset the Game"}
    </button>
    {winners(board) !== null && <div id="win">{`Winner: Player ${winners(board)}!`}</div>}
    {board.every(square => square !== null) && <div id="draw">{`It's a draw!`}</div>}
  </>
  );

}
